221 template <typename T>
222 inline T log(T x){
223 if(x<=0) throw EexceptionArithemetic("Log of non-positive number", "shed_math::log");
224
225 // ln(x) = 2 * sum ( ((x-1)/(x+1))^(2n+1) / (2n+1) )
226 T y = (x - static_cast<T>(1)) / (x + static_cast<T>(1));
227 T y2 = y * y;
228 T res = y;
229 T term = y;
230 for(int i = 1;i<50;++i){
231 term *= y2;
232 res += term / static_cast<T>(2*i+1);
233 if(abs(term) < static_cast<T>(1e-12)) break;
234 }
235 return static_cast<T>(2)*res;
236 }

267 inline float64 pow(float64 base,float64 exp_val){
268 if(exp_val == 0) return 1.0; // 任何數的0次方都是0
269 if(is_nan(exp_val)) return exp_val;//指數為NaN返回自己
270 if(base == 1.0) return 1.0;
271 if(base == 0.0){
272 if(exp_val <0) return -1.0 / 0.0;//inf
273 else if(exp_val == 0) return 1;
274 else return 0;
275 }
276
277 // 如果是整數冪次，使用快速冪
278 if(exp_val == trunc(exp_val)){
279 return pow<float64,int64>(base,static_cast<int64>(exp_val));
280 }
281
282 if(base <0.0){
283 // 負數的小數次方是NaN
284 return 0.0 / 0.0;
285 }
286
287 return exp(exp_val * log(base));
288 }

290 template <typename T>
291 inline enable_if_type<!is_integral<T>::value,T> pow(T base,T exp_val){
292 // 1. 處理底數為 0 的情況
293 if (base == static_cast<T>(0)) {
294 if (exp_val > static_cast<T>(0)) return static_cast<T>(0);
295 if (exp_val == static_cast<T>(0)) return static_cast<T>(1); // 0^0 定義為 1
296 return static_cast<T>(1) / static_cast<T>(0); // 0 的負次方 = INF
297 }
298
299 // 2. 處理負底數
300 if (base < static_cast<T>(0)) {
301 return static_cast<T>(0) / static_cast<T>(0);
302 }
303
304 // 3. a^b = exp(b * ln(a))
305 return exp(exp_val * log(base));
306 }
