50 class IiostreamBufConsoleInput : public IiostreamBuf {
51 private:
52 static const int INNER_BUF_SIZE = 1024; // 输入缓冲区大小
53 char _input_buf[INNER_BUF_SIZE]; // 仅用于输入的缓冲区
54 int _input_pos = 0; // 输入读取位置（当前已读到的位置）
55 int _input_len = 0; // 输入缓冲区中有效数据长度
56
57 // 从控制台读取数据填充输入缓冲区
58 void _fill_input_buf() {
59 #ifdef _WIN32
60 HANDLE hConsole = GetStdHandle(STD_INTPUT_HANDLE);
61 if (hConsole == INVALID_HANDLE_VALUE) {
62 setState(Iiostream_state::IO_BAD);
63 _input_len = 0;
64 return;
65 }
66 DWORD read = 0;
67 if (!ReadConsoleA(hConsole, _input_buf, INNER_BUF_SIZE, &read, nullptr)) {
68 setState(Iiostream_state::IO_FAIL);
69 _input_len = 0;
70 return;
71 }
72 _input_len = static_cast<int>(read);
73 #else
74 int read_bytes = read(STDIN_FILENO, _input_buf, INNER_BUF_SIZE);
75 if (read_bytes < 0) {
76 setState(Iiostream_state::IO_FAIL);
77 _input_len = 0;
78 return;
79 }
80 _input_len = read_bytes;
81 #endif
82 _input_pos = 0; // 重置读取位置
83 }
84
85 public:
86 ~IiostreamBufConsoleInput() override = default; // 输入缓冲区无需刷新
87
88 // 输入缓冲区不支持输出操作，调用则设置错误状态
89 void flush() override {}
90 void stream_put_char(char) override { setState(Iiostream_state::IO_BAD); }
91 void stream_put_n_char(const char*, int) override { setState(Iiostream_state::IO_BAD); }
92
93 // 输入接口：获取下一个字符并移动指针
94 int stream_bump_char() override {
95 if (_input_pos >= _input_len) {
96 _fill_input_buf(); // 输入缓冲区空了，填充新数据
97 if (_input_len == 0) {
98 setState(Iiostream_state::IO_EOF);
99 return -1; // EOF
100 }
101 }
102 return static_cast<unsigned char>(_input_buf[_input_pos++]);
103 }
104
105 // 输入接口：获取当前字符（不移动指针）
106 int stream_get_char() override {
107 if (_input_pos >= _input_len) {
108 _fill_input_buf(); // 输入缓冲区空了，填充新数据
109 if (_input_len == 0) {
110 setState(Iiostream_state::IO_EOF);
111 return -1; // EOF
112 }
113 }
114 return static_cast<unsigned char>(_input_buf[_input_pos]);
115 }

118 class IiostreamBufConsoleOutput : public IiostreamBuf {
119 private:
120 static const int INNER_BUF_SIZE = 1024; // 输出缓冲区大小
121 char _output_buf[INNER_BUF_SIZE]; // 仅用于输出的缓冲区
122 int _output_pos = 0; // 输出写入位置（当前已写到的位置）
123
124 // 刷新输出缓冲区到控制台
125 void _flush_output() {
126 if (_output_pos <= 0) return; // 缓冲区为空，无需刷新
127
128 #ifdef _WIN32
129 HANDLE hConsole = GetStdHandle(STD_OUTPUT_HANDLE);
130 if (hConsole == INVALID_HANDLE_VALUE) {
131 setState(Iiostream_state::IO_BAD);
132 return;
133 }
134 DWORD written = 0;
135 if (!WriteConsoleA(hConsole, _output_buf, _output_pos, &written, nullptr)) {
136 setState(Iiostream_state::IO_FAIL);
137 }
138 #else
139 int written = write(STDOUT_FILENO, _output_buf, _output_pos);
140 if (written < 0) {
141 setState(Iiostream_state::IO_FAIL);
142 }
143 #endif
144 _output_pos = 0; // 重置写入位置
145 }
146
147 public:
148 ~IiostreamBufConsoleOutput() override {
149 _flush_output(); // 析构时确保缓冲区数据输出
150 }
151
152 // 刷新输出缓冲区
153 void flush() override {
154 _flush_output();
155 }
156
157 // 输出接口：写入单个字符（满了自动刷新）
158 void stream_put_char(char c) override {
159 if (_output_pos >= INNER_BUF_SIZE) {
160 _flush_output(); // 缓冲区满了，先刷新
161 }
162 _output_buf[_output_pos++] = c;
163 }
164
165 // 输出接口：写入指定长度字符串（分批处理）
166 void stream_put_n_char(const char* str, int n) override {
167 if (!str || n <= 0) return;
168
169 while (n > 0) {
170 int remain = INNER_BUF_SIZE - _output_pos; // 缓冲区剩余空间
171 if (remain <= 0) {
172 _flush_output(); // 空间不足，先刷新
173 remain = INNER_BUF_SIZE;
174 }
175 // 复制最小剩余空间和剩余字符数
176 int copy_len = (n < remain) ? n : remain;
177 for (int i = 0; i < copy_len; ++i) {
178 _output_buf[_output_pos++] = str[i];
179 }
180 str += copy_len;
181 n -= copy_len;
182 }
183 }
184
185 // 输出缓冲区不支持输入操作，调用则设置错误状态
186 int stream_bump_char() override {
187 setState(Iiostream_state::IO_BAD);
188 return -1;
189 }
190
191 int stream_get_char() override {
192 setState(Iiostream_state::IO_BAD);
193 return -1;
194 }
195 };
