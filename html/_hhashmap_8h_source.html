22template <typename K>
23struct DefaultHash {
24 static constexpr unsigned long long C1 = 0xff51afd7ed558ccdULL; // 哈希常数1
25 static constexpr unsigned long long C2 = 0xc4ceb9fe1a85ec53ULL; // 哈希常数2
26 static constexpr unsigned long long R1 = 33; // 右移位数
27
33 int operator()(const K& key) const {
34 unsigned long long h = static_cast<unsigned long long>(key);
35
36 h ^= h >> R1;
37 h *= C1;
38 h ^= h >> R1;
39 h *= C2;
40 h ^= h >> R1;
41
42 return static_cast<int>(h & 0xFFFFFFFF); // 截取低32位
43 }
44};
45
49template <>
50struct DefaultHash<Sstring> {
51 static const int PRIME = 31; // 哈希质数
52
58 int operator()(const Sstring& key) const {
59 if (key.size() == 0) {
60 return 0;
61 }
62 int hash = 0;
63 for (int i = 0; i < key.size(); ++i) {
64 // 转换为无符号字符避免符号扩展问题
65 hash = PRIME * hash + static_cast<int>(static_cast<unsigned char>(key[i]));
66 }
67 return hash;
68 }
69};
