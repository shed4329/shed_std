23template <typename K>
24struct DefaultHash {
25 static constexpr unsigned long long C1 = 0xff51afd7ed558ccdULL; // 哈希常数1
26 static constexpr unsigned long long C2 = 0xc4ceb9fe1a85ec53ULL; // 哈希常数2
27 static constexpr unsigned long long R1 = 33; // 右移位数
28
34 int operator()(const K& key) const {
35 unsigned long long h = static_cast<unsigned long long>(key);
36
37 h ^= h >> R1;
38 h *= C1;
39 h ^= h >> R1;
40 h *= C2;
41 h ^= h >> R1;
42
43 return static_cast<int>(h & 0xFFFFFFFF); // 截取低32位
44 }
45};
46
50template <>
51struct DefaultHash<Sstring> {
52 static const int PRIME = 31; // 哈希质数
53
59 int operator()(const Sstring& key) const {
60 if (key.size() == 0) {
61 return 0;
62 }
63 int hash = 0;
64 for (int i = 0; i < key.size(); ++i) {
65 // 转换为无符号字符避免符号扩展问题
66 hash = PRIME * hash + static_cast<int>(static_cast<unsigned char>(key[i]));
67 }
68 return hash;
69 }
70};
