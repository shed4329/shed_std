398 int Aarray<E>::find_last(const E& value,int start_index,int end_index) const{
399 // 1. 范围合法性检查
400 if (start_index < 0 || end_index > this->_length || start_index >= end_index) {
401 return -1; // 无效范围或空范围，直接返回-1
402 }
403 // 2. 从end_index-1开始反向遍历到start_index
404 Aarray_const_iterator curr_it(end_index - 1, this);
405 Aarray_const_iterator start_it(start_index, this);
406 while (true) {
407 if (*curr_it == value) {
408 return curr_it - begin(); // 找到匹配值
409 }
410 if (curr_it == start_it) {
411 break; // 已遍历到起始位置，退出循环
412 }
413 --curr_it; // 继续向前移动
414 }
415 // 3. 未找到匹配值
416 return -1;
417 }
