38namespace shed_std{
39 class IiostreamBufConsole:public IiostreamBuf{
40 private:
41 static const int INNER_BUF_SIZE = 1024; // 缓存大小
42 char _inner_buf[INNER_BUF_SIZE]; // 缓冲区
43 int _buf_pos = 0; // 当前缓存的写入位置
44 public:
45 ~IiostreamBufConsole() override{
46 flush(); //销毁前刷新缓冲区
47 }
48
49 // 刷新缓存:调用系统API输出到控制台
50 void flush() override{
51 if(_buf_pos <= 0){
52 return;
53 }
54
55 #ifdef _WIN32
56 // Windows:用extern声明的API输出
57 HANDLE hConsole = GetStdHandle(STD_OUTPUT_HANDLE);
58 if(hConsole == INVALID_HANDLE_VALUE){
59 setState(Iiostream_state::IO_BAD);
60 return;
61 }
62 DWORD written = 0;
63 // 调用WriteConsoleA 输出缓存内容
64 if(!WriteConsoleA(hConsole,_inner_buf,_buf_pos,&written,nullptr)){
65 setState(Iiostream_state::IO_FAIL);
66 }
67 # else
68 // Linux/macOS: 调用write进行系统调用
69 int written = write(STDOUT_FILENO,_inner_buf,_buf_pos);
70 if(written <0){
71 setState(Iiostream_state::IO_FAIL);
72 }
73 #endif
74
75 _buf_pos = 0; //重置缓存位置
76 }
77
78 // 输出单个字符:先存缓存,满了刷新
79 void stream_put_char(char c) override{
80 if(_buf_pos >= INNER_BUF_SIZE){
81 flush();
82 }
83 _inner_buf[_buf_pos++] = c;
84 }
85
86 // 输出指定长度字符串:分批写入缓存
87 void stream_put_n_char(const char* str,int n) override{
88 if(!str || n<=0) return;
89
90 while(n>0){
91 int remain = INNER_BUF_SIZE - _buf_pos;
92 // 该刷新了
93 if(remain <=0){
94 flush();
95 remain = INNER_BUF_SIZE;
96 }
97 // 复制到缓存(取剩余缓存和剩余字符串长度中的最小值)
98 int copy_len = (n<remain)?n:remain;
99 for(int i=0;i<copy_len;++i){
100 _inner_buf[_buf_pos++] = str[i];
101 }
102 str += copy_len;
103 n -= copy_len;
104 }
105 }
106
107 // 输入接口:返回EOF即可，输入流无需实现
108 int stream_bump_char() override{
109 setState(Iiostream_state::IO_EOF);
110 return -1;
111 }
112
113 int stream_get_char() override{
114 setState(Iiostream_state::IO_EOF);
115 return -1;
116 }
117 };
118}
