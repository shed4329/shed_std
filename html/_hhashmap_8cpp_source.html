16 template <typename K>
17 struct DefaultHash{
18 static constexpr unsigned long long C1 = 0xff51afd7ed558ccdULL;
19 static constexpr unsigned long long C2 = 0xc4ceb9fe1a85ec53ULL;
20 static constexpr unsigned long long R1 = 33;// 右移位数
21
22 int operator()(const K& key) const{
23 unsigned long long h = (unsigned long long) key;
24
25 h ^= h >> R1;
26 h *= C1;
27 h ^= h >> R1;
28 h *= C2;
29 h ^= h >> R1;
30
31 // 保持用户定义的 AND 操作
32 return (int)(h & 0xFFFFFFFF);
33 }
34 };

39 template <>
40 struct DefaultHash<Sstring>{
41 static const int PRIME = 31;
42
43 int operator()(const Sstring& key) const{
44 if(key.size() == 0){
45 return 0;
46 }
47 int hash = 0;
48
49 for(int i = 0 ; i<key.size() ; ++i){
50 // 确保字符转换为 int 是正确的
51 hash = PRIME * hash + (int)((unsigned char)key[i]);
52 }
53 return hash;
54 }
55 };

72 void _auto_expand(){
73 // 使用 size() 获取 Aarray 的长度（桶数量）
74 int current_capacity = _data.size();
75
76 // 初始容量检查 (防止 _data 长度为 0 导致 get_index 失败)
77 if (current_capacity == 0) {
78 // 强制初始化到默认容量
79 rehash(DEFAULT_INITIAL_CAPACITY);
80 return;
81 }
82
83 // 负载因子检查: size > capacity * LOAD_FACTOR
84 if ((double)_size > (double)current_capacity * LOAD_FACTOR) {
85 int new_capacity = current_capacity * 2;
86
87 if (new_capacity <= 0) {
88 return;

91 rehash(new_capacity);
92 }

98 void rehash(int new_capacity){
99 // 使用 size() 代替 capacity()
100 if(new_capacity > _data.size()){
101 // 1. 创建新数组，使用 Aarray(N) 构造函数初始化正确的长度
102 // Aarray 的元素（Llist）会使用默认构造函数初始化
103 Aarray<Llist<pair<K,V>>> new_buckets(new_capacity);
104
105 // 2. 旧桶移动到新桶
106 int old_capacity = _data.size(); // 使用 size() 代替 capacity()
107 for(int i = 0;i<old_capacity;++i){
108 Llist<pair<K,V>>& old_list = _data[i];
109
110 // 使用迭代器遍历链表
111 for(typename Llist<pair<K,V>>::Llist_iterator it = old_list.begin(); it != old_list.end(); ++it){
112 pair<K,V> p = *it;
113
114 // 计算新索引 (基于新的 capacity)
115 int hash_val = (int)_hash(p.first);
116 // 由于 new_capacity 是2的幂次，使用位运算取模
117 int new_index = hash_val & (new_capacity - 1);
118
119 // 插入到新桶中
120 new_buckets[new_index].add_at_front(p);
121 }
122 }
123
124 // 3. 替换桶数组。旧的 _data 将通过赋值运算符安全析构。
125 _data = new_buckets;
126 }
127 }
